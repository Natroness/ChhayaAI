rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isOwner(uid) {
      return signedIn() && request.auth.uid == uid;
    }

    function isAcceptedFriendOf(uid) {
      return signedIn() &&
        exists(/databases/$(database)/documents/users/$(uid)/friends/$(request.auth.uid));
    }

    // User profiles
    match /users/{uid} {
      allow read: if signedIn();
      allow write: if isOwner(uid);
    }

    // Friend subcollections
    match /users/{uid}/friends/{friendUid} {
      allow read: if isOwner(uid);
      allow write: if isOwner(uid) || isOwner(friendUid);
    }

    // Friend requests
    match /friend_requests/{requestId} {
      allow read: if signedIn() &&
        (resource.data.fromUid == request.auth.uid || resource.data.toUid == request.auth.uid);
      allow create: if signedIn() && request.resource.data.fromUid == request.auth.uid;
      allow update: if signedIn() &&
        (resource.data.fromUid == request.auth.uid || resource.data.toUid == request.auth.uid);
    }

    // Live locations
    match /live_locations/{uid} {
      allow read: if isOwner(uid) || isAcceptedFriendOf(uid);
      allow write: if isOwner(uid);
    }

    // Emergency operators (read-only for clients, written by seed script / admin SDK)
    match /emergency_operators/{operatorId} {
      allow read: if signedIn();
      allow write: if false;
    }

    // SOS events
    match /sos_events/{sosId} {
      allow read: if signedIn();
      allow create: if signedIn();
      allow update: if signedIn() && resource.data.createdByUid == request.auth.uid;
    }

    // SOS responders
    match /sos_events/{sosId}/responders/{responderId} {
      allow read: if signedIn();
      allow write: if false;
    }

    // Alert items
    match /alerts/{uid}/items/{alertId} {
      allow read: if isOwner(uid);
      allow write: if false;
    }
  }
}
